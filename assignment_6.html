#include <stdio.h>
#include <stdlib.h>

typedef struct stack {
    int data;
    struct stack *top;
} *STACK;

STACK push_ll(STACK stack, int data) {
    STACK new_node = (STACK)malloc(sizeof(struct stack));
    if (new_node == NULL) {
        printf("\nOverflow");
        return stack;
    }
    new_node->data = data;
    new_node->top = NULL;
    if (stack == NULL) {
        stack = new_node;
    } else {
        STACK ptr = stack;
        while (ptr->top!=NULL) {
            ptr = ptr->top;
        }
        ptr->top = new_node;
    }
    return stack;
}

STACK pop_ll(STACK stack) {
    if (stack == NULL) {
        printf("\nUnderflow");
    } else if (stack->top == NULL) {
        stack = NULL;
    } else {
        STACK ptr = stack;
        while (ptr->top->top!=NULL) {
            ptr = ptr->top;
        }
        ptr->top = NULL;
    }
    return stack;
}

void seek_ll(STACK stack) {
    STACK ptr = stack;
    if (stack == NULL) {
        printf("\nUnderflow");
    } else {
        while (ptr->top!=NULL) {
            ptr = ptr->top;
        }
        printf("Top: %d\n", ptr->data);
    }
}


int push(int arr[], int element, int top, int size) {
    if (top<size-1) {
        top++;
        arr[top] = element;
    } else {
        printf("\nOVERFLOW %d\n", element);
    }
    return top;
}

int pop(int arr[], int top) {
    if (top!=-1) {
        printf("%d\n", arr[top]);
        top--;
    } else {
        printf("\nUNDERFLOW\n");
    }
    return top;
}

void seek(int arr[], int top) {
    if (top != -1)
        printf("\ntop element: %d\n", arr[top]);
    else 
        printf("\nUnderflow\n");
}
