WAP to read, display, add and subtract two complex numbers. (solve it using structure and pointer)
*/

#include <stdio.h>

typedef struct {
    int real, img;
} complexNum;

complexNum add(complexNum *x, complexNum *y) {
    complexNum res;
    res.real = x->real + y->real;
    res.img = x->img + y->img;
    return res;
}

complexNum subtract(complexNum *x, complexNum *y) {
    complexNum res;
    res.real = x->real - y->real;
    res.img = x->img - y->img;
    return res;
}

void main() {
    complexNum x, y, a, s;
    printf("a + bi\nEnter value of a and b for 1st complex number: ");
    scanf("%d%d", &x.real, &x.img);
    printf("Enter value of a and b for 2nd complex number: ");
    scanf("%d%d", &y.real, &y.img);
    int z;
    printf("Press 1 to add the numbers\nPress 2 to subtract the numbers\nYour input: ");
    scanf("%d", &z);
    if (z == 1) {
        a = add(&x, &y);
        printf("(%d + %di) + (%d + %di) = (%d + %di)\n", x.real, x.img, y.real, y.img, a.real, a.img);
    } else if (z == 2) {
        s = subtract(&x, &y);
        printf("(%d + %di) - (%d + %di) = (%d + %di)", x.real, x.img, y.real, y.img, s.real, s.img);
    } else {
        printf("Invalid operator");
    }
}

Declare a structure fraction that has two fields-numerator and denominator. Create two variables and compare them using function. Return 0 if the two fractions are equal, -1 if the first fraction is less than the second, and 1 otherwise. You may convert a fraction into a floating point number for your convenience.
/*
Declare a structure fraction that has two fields— numerator and denominator. 
Create two variables and compare them using function. 
Return 0 if the two fractions are equal, –1 if the first fraction is less than the second, and 1 otherwise. 
You may convert a fraction into a floating point number for your convenience.
*/
#include <stdio.h>

typedef struct {
    int n, d;
} div;

int compare(div a, div b) {
    float n1 = a.n/(a.d+0.0), n2 = b.n/(b.d+0.0);
    if (n1 == n2) {
        return 0;
    } else if (n1<n2) {
        return -1;
    } else {
        return 1;
    }
}

void main() {
    div a, b;
    printf("Enter the first numerator and denominator: ");
    scanf("%d%d", &a.n, &a.d);
    printf("Enter the second numerator and denominator: ");
    scanf("%d%d", &b.n, &b.d);
    int z = compare(a, b);
    if (z == 0) {
        printf("both are equal");
    } else if (z == -1) {
        printf("1st is smaller");
    } else if (z == 1) {
        printf("1st is greater");
    }
}
