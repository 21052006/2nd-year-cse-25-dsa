#include <stdio.h>
#include <stdlib.h>

typedef struct node {
    int data;
    struct node *lptr;
    struct node *rptr;
} *NODE;

NODE insert_at_first(NODE start, int val) {
    NODE new_node = (NODE)malloc(sizeof(struct node));
    if (new_node == NULL) {
        printf("OVERFLOW");
        return start;
    }
    new_node->data = val;
    if (start == NULL) {
        start = new_node;
        start->lptr = NULL;
        start->rptr = NULL;
    } else {
        new_node->rptr = start;
        new_node->lptr = NULL;
        start->lptr = new_node;
        start = new_node;
    }
}

NODE insert_at_last(NODE start, int val) {
    NODE new_node = (NODE)malloc(sizeof(struct node));
    if (new_node == NULL) {
        printf("OVERFLOW");
        return start;
    }
    new_node->data = val;
    NODE ptr = start;
    while (ptr->rptr!=NULL) {
        ptr = ptr->rptr;
    }
    new_node->rptr = NULL;
    new_node->lptr = ptr;
    ptr->rptr = new_node;
    return start;
}

NODE insert_at_given_pos(NODE start, int val, int pos) {
    NODE new_node = (NODE)malloc(sizeof(struct node));
    if (new_node == NULL) {
        printf("OVERFLOW");
        return start;
    }
    new_node->data = val;
    if (start == NULL || pos == 1) {
        new_node->lptr = NULL;
        new_node->rptr = start;
        start = new_node;
    } else {
        NODE ptr = start;
        int c = 0;
        while (c<pos-1) {
            c++;
            ptr = ptr->rptr;
            if (ptr == NULL) {
                printf("Index Out of Range\n\n");
                return start;
            }
        }
        if (ptr!=NULL) {
            new_node->lptr = ptr->lptr;
            new_node->rptr = ptr;
            ptr->lptr->rptr = new_node;
            ptr->lptr = new_node;
        }
        return start;
    }
}

NODE delete_first(NODE start) {
    if (start == NULL) {
        printf("UNDERFLOW\n\n");
        return start;
    }
    NODE ptr = start;
    start = start->rptr;
    if (start != NULL) {
        start->lptr = NULL;
        ptr->rptr = NULL;
    }
    free(ptr);
    return start;
}

NODE delete_last(NODE start) {
    if (start == NULL) {
        printf("UNDERFLOW\n\n");
        return start;
    }
    NODE ptr = start;
    if (start->rptr == NULL) {
        start = NULL;
    } else {
        while (ptr->rptr != NULL) {
            ptr = ptr->rptr;
        }
        ptr->lptr->rptr = NULL;
        ptr->lptr = NULL;
        free(ptr);
    }
    return start;
}

void traverse(NODE start) {
    if (start == NULL) {
        printf("UNDERFLOW\n");
        return;
    }
    NODE ptr = start;
    while (ptr!=NULL) {
        printf("%d ", ptr->data);
        ptr = ptr->rptr;
    }
    printf("\n");
}
