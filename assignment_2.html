}

WAP to find the largest number and count the occurrence of the largest number in an array of n integers using a single loop.

void main() {
    int n, c = 1, m;
    printf("Enter the length of array: ");
    scanf("%d", &n);
    printf("Enter the elements of array: ");
    int arr[n];
    for (int i = 0;i<n;i++) {
        scanf("%d", &arr[i]);
    }
    m = arr[0];
    for (int i = 1;i<n;i++) {
        if (arr[i]>m) {
            m = arr[i];
            c = 0;
        }
        if (arr[i] == m) {
            c++;
        }
    }
    printf("Largest element is %d and occurs %d times", m, c);
}

WAP to reverse the contents of an array of n elements.
#include <stdio.h>

void reverse(int n, int *arr) {
    for (int i = 0;i<(n/2);i++) {
        int temp = *(arr+i);
        *(arr+i) = *(arr+n-i-1);
        *(arr+n-i-1) = temp;
    }
}

void main() {
    int n;
    printf("Enter the length of array: ");
    scanf("%d", &n);
    printf("Enter the elements of array: ");
    int arr[n];
    for (int i = 0;i<n;i++) {
        scanf("%d", &arr[i]);
    }
    reverse(n, arr);
    for (int i = 0;i<n;i++) {
        printf("%d ", arr[i]);
    }
}

        int flag = 1;
